using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace TravellingAtTheSpeedOfSound
{
    class Program
    {
        static void Main(string[] args)
        {
		//Angel Terziev
            int Velocity = 1225;
            int distance = 149600000;
            double Time = 0;
            double Years = 0;
            double Days = 0;
            Time = distance / Velocity;
            Years = Math.Floor(Time / 365);
            Days = Math.Floor(Time % 365);
            Console.WriteLine("The time needed to travel from the Eath to the Sun is {0} hours, or {1} Years and {2} days", Time, Years, Days);

        }
    }
}
