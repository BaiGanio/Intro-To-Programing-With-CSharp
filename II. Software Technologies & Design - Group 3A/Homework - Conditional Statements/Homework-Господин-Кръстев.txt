namespace TravellingToTheSunAtTheSpeedOfSound
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Do you want to know how much time does it take for an object to travel from the Earth to the Sun at the speed of light?");
            Console.WriteLine("Answer y/n");
            string answer =  Console.ReadLine();


            if (answer == "y")
            {
                int distanceToSun = 149600000;
                int speedOfSoundPerHour = 490;
                // Speed and Distance are approximate 
                double traveltimehours = distanceToSun / speedOfSoundPerHour;
                double traveltimeyears = (traveltimehours / 24) / 365;

                Console.WriteLine("The time it takes for an object to travel from the Earth to the Sun at the speed of sound is approximately {0} hours", traveltimehours);
                Console.WriteLine("The time it takes for an object to travel from the Earth to the Sun at the speed of sound is approximately {0} years", traveltimeyears);
                Console.WriteLine("Thank you for answering correctly and Goodbye! ");

            }
            else if (answer == "n") { Console.WriteLine("Thank you for answering correctly! Have a nice day!"); }
            else { Console.WriteLine("Goodbye !"); }

        }
    }
}
