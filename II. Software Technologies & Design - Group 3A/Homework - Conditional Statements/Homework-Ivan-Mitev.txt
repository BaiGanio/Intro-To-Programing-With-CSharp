namespace Peoplesweight
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Please enter your weight: ");

            Console.Write("Please enter the planet: ");
            string planet = (Console.ReadLine());

            switch(planet)
            {
                case "Mercury" :
                    weight = weight * 0.38;
                    Console.WriteLine("Your weight on Mercury would be : {0}",weight);
                    break;
                case "Venus":
                    weight = weight * 0.91;
                    Console.WriteLine("Your weight on Venus would be : {0}", weight);
                    break;
                case "Earth":
                    weight = weight * 1;
                    Console.WriteLine("Your weight on Earth is : {0}", weight);
                    break;
                case "Mars":
                    weight = weight * 0.38;
                    Console.WriteLine("Your weight on Mars would be : {0}", weight);
                    break;
                case "Jupiter":
                    weight = weight * 2.54;
                    Console.WriteLine("Your weight on Jupiter would be : {0}", weight);
                    break;
                case "Saturn":
                    weight = weight * 1.08;
                    Console.WriteLine("Your weight on Saturn would be : {0}", weight);
                    break;
                case "Uranus":
                    weight = weight * 0.91;
                    Console.WriteLine("Your weight on Uranus would be : {0}", weight);
                    break;
                case "Neptune":
                    weight = weight * 1.19;
                    Console.WriteLine("Your weight on Neptune would be : {0}", weight);
                    break;
                case "Pluto":
                    weight = weight * 0.06;
                    Console.WriteLine("Your weight on Pluto would be : {0}", weight);
                    break;
                default:
                    Console.WriteLine("Please enter the name correctly!"); 
                    break;
            }
            
        }
    }
}
